include_directories(${XROOTD_INCLUDE_DIRS})

#-------------------------------------------------------------------------------
# FuseCache library
#-------------------------------------------------------------------------------
add_library(
  FuseCache SHARED
  FuseWriteCache.cc  FuseWriteCache.hh
  CacheEntry.cc      CacheEntry.hh
  FileAbstraction.cc FileAbstraction.hh
  LayoutWrapper.cc   LayoutWrapper.hh
  ../xrdutils.cc     ../xrdutils.hh)

target_link_libraries(
  FuseCache PUBLIC
  EosFstIo
  ${XROOTD_CL_LIBRARY}
  ${XROOTD_UTILS_LIBRARY}
  ${CMAKE_THREAD_LIBS_INIT})

target_compile_definitions(
  FuseCache PUBLIC
  -D_LARGEFILE_SOURCE -D_LARGEFILE64_SOURCE -D_FILE_OFFSET_BITS=64)

set_target_properties(
  FuseCache
  PROPERTIES
  VERSION ${VERSION}
  SOVERSION ${VERSION_MAJOR}
  MACOSX_RPATH TRUE)

if (Linux)
  add_library(
    FuseCache-Static STATIC
    FuseWriteCache.cc  FuseWriteCache.hh
    CacheEntry.cc      CacheEntry.hh
    FileAbstraction.cc FileAbstraction.hh
    LayoutWrapper.cc LayoutWrapper.hh)

  target_link_libraries(
    FuseCache-Static PUBLIC
    EosFstIo-Static
    ${XROOTD_CL_LIBRARY}
    ${XROOTD_UTILS_LIBRARY}
    ${CMAKE_THREAD_LIBS_INIT})

  target_compile_definitions(
    FuseCache-Static PUBLIC
    -D_LARGEFILE_SOURCE -D_LARGEFILE64_SOURCE -D_FILE_OFFSET_BITS=64)

  set_target_properties(
    FuseCache-Static PROPERTIES
    POSITION_INDEPENDENT_CODE TRUE)
else()
install(
  TARGETS FuseCache
  LIBRARY DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR}
  ARCHIVE DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR})
endif()
