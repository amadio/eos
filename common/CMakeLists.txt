#-------------------------------------------------------------------------------
# Build the libfmt for fast convertions
#-------------------------------------------------------------------------------
add_subdirectory(fmt)

#-------------------------------------------------------------------------------
# Generate protocol buffer files
#-------------------------------------------------------------------------------
PROTOBUF_GENERATE_CPP(DBMAPTEST_SRCS DBMAPTEST_HDRS PROTOFILES dbmaptest/test.proto)

  ${DBMAPTEST_SRCS}
  ${DBMAPTEST_HDRS}
  PROPERTIES GENERATED 1)

#-------------------------------------------------------------------------------
# eosCommon library
#-------------------------------------------------------------------------------
set(
  EOSCOMMON_SRCS
  Fmd.cc
  Mapping.cc
  Statfs.cc
  SymKeys.cc
  GlobalConfig.cc
  Report.cc
  StringTokenizer.cc
  StringConversion.cc
  CommentLog.cc
  RWMutex.cc
  JeMallocHandler.cc)

add_library(eosCommon SHARED ${EOSCOMMON_SRCS})

target_link_libraries(
  eosCommon PUBLIC
  XrdMqClient
  fmt
  ${Z_LIBRARY}
  ${UUID_LIBRARIES}
  ${ATTR_LIBRARIES}
  ${NCURSES_LIBRARY}
  ${CURL_LIBRARIES}
  ${XROOTD_CL_LIBRARY}
  ${XROOTD_UTILS_LIBRARY}
  ${OPENSSL_CRYPTO_LIBRARY})

# !!!!!
# TODO: the HAVE_ATOMICS flags should be removed and the code in RWMutex should
# use std::atomic and the <atomic> header.
# !!!!!
target_compile_definitions(
  eosCommon
  PUBLIC -DDAEMONUID=${DAEMONUID} -DDAEMONGID=${DAEMONGID} -DHAVE_ATOMICS=1)

target_include_directories(
  eosCommon PUBLIC ${CURL_INCLUDE_DIRS})

set_target_properties(
  eosCommon PROPERTIES
  VERSION ${VERSION}
  SOVERSION ${VERSION_MAJOR}
  MACOSX_RPATH TRUE)

if(Linux)
  target_link_libraries(
    eosCommon PUBLIC
    ${GLIBC_RT_LIBRARY})
endif()

install(
  TARGETS eosCommon
  LIBRARY DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR}
  RUNTIME DESTINATION ${CMAKE_INSTALL_FULL_BINDIR}
  ARCHIVE DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR})

add_library(eosCommon-Static STATIC ${EOSCOMMON_SRCS})

target_include_directories(
  eosCommon-Static PUBLIC
  $<BUILD_INTERFACE:${PROJECT_SOURCE_DIR};${CURL_INCLUDE_DIRS}>
  $<INSTALL_INTERFACE:include>)

# !!!!!
# TODO: the HAVE_ATOMICS flags should be removed and the code in RWMutex should
# use std::atomic and the <atomic> header.
# !!!!!
target_compile_definitions(
  eosCommon-Static
  PUBLIC -DDAEMONUID=${DAEMONUID} -DDAEMONGID=${DAEMONGID} -DHAVE_ATOMICS=1)

set_target_properties(
  eosCommon-Static PROPERTIES
  POSITION_INDEPENDENT_CODE TRUE)

target_link_libraries(
  eosCommon-Static PRIVATE
  XrdMqClient-Static
  fmt
  ${Z_LIBRARY}
  ${UUID_LIBRARIES}
  ${ATTR_LIBRARIES}
  ${NCURSES_LIBRARY}
  ${CURL_LIBRARIES}
  ${XROOTD_CL_LIBRARY}
  ${XROOTD_UTILS_LIBRARY}
  ${OPENSSL_CRYPTO_LIBRARY}
  ${CMAKE_THREAD_LIBS_INIT})
